def standardize(train_df, val_df, test_df):
    mean = train_df.mean(axis=0)
    std = train_df.std(axis=0).replace(0, 1.0)

    train_scaled = (train_df - mean) / std
    val_scaled = (val_df - mean) / std
    test_scaled = (test_df - mean) / std

    return train_scaled, val_scaled, test_scaled, mean, std


def make_dataset(data_df, target_col, sequence_length, sampling_rate, delay_rows, batch_size, shuffle):
    data = data_df.values.astype("float32")
    target_idx = data_df.columns.get_loc(target_col)

    max_start = len(data) - delay_rows - (sequence_length - 1) * sampling_rate
    if max_start <= 0:
        raise ValueError("Invalid sequence_length / sampling_rate / delay_rows combination")

    targets = data[delay_rows + (sequence_length - 1) * sampling_rate:, target_idx]
    inputs = data[:len(data) - delay_rows]

    ds = keras.utils.timeseries_dataset_from_array(
        data=inputs,
        targets=targets,
        sequence_length=sequence_length,
        sampling_rate=sampling_rate,
        batch_size=batch_size,
        shuffle=shuffle,
        start_index=0,
        end_index=len(inputs) - 1
    )
    return ds


def inverse_target(y_scaled, mean, std, target_col):
    return y_scaled * std[target_col] + mean[target_col]


def evaluate_model_original_mae(model, dataset, mean, std, target_col):
    y_true_scaled = []
    y_pred_scaled = []

    for x_batch, y_batch in dataset:
        pred = model.predict(x_batch, verbose=0).reshape(-1)
        y_true_scaled.append(y_batch.numpy().reshape(-1))
        y_pred_scaled.append(pred)

    y_true_scaled = np.concatenate(y_true_scaled)
    y_pred_scaled = np.concatenate(y_pred_scaled)

    y_true = inverse_target(y_true_scaled, mean, std, target_col)
    y_pred = inverse_target(y_pred_scaled, mean, std, target_col)

    mae = np.mean(np.abs(y_true - y_pred))
    return mae, y_true, y_pred


def baseline_mae_original(data_df, target_col, sequence_length, sampling_rate, delay_rows, mean, std):
    data = data_df.values.astype("float32")
    target_idx = data_df.columns.get_loc(target_col)

    max_start = len(data) - delay_rows - (sequence_length - 1) * sampling_rate
    preds = []
    trues = []

    for start in range(max_start):
        last_input_idx = start + (sequence_length - 1) * sampling_rate
        target_row_idx = start + delay_rows + (sequence_length - 1) * sampling_rate

        preds.append(data[last_input_idx, target_idx])
        trues.append(data[target_row_idx, target_idx])

    preds = np.array(preds)
    trues = np.array(trues)

    preds = inverse_target(preds, mean, std, target_col)
    trues = inverse_target(trues, mean, std, target_col)

    return np.mean(np.abs(trues - preds))

def build_single_lstm(input_shape, units=32, dropout=0.2, recurrent_dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.LSTM(units, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.Dense(1)
    ])


def build_single_gru(input_shape, units=32, dropout=0.2, recurrent_dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.GRU(units, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.Dense(1)
    ])


def build_stacked_lstm(input_shape, units1=32, units2=32, dropout=0.2, recurrent_dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.LSTM(units1, return_sequences=True, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.LSTM(units2, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.Dense(1)
    ])


def build_stacked_gru(input_shape, units1=32, units2=64, dropout=0.2, recurrent_dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.GRU(units1, return_sequences=True, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.GRU(units2, dropout=dropout, recurrent_dropout=recurrent_dropout),
        layers.Dense(1)
    ])


def build_cnn_gru(input_shape, conv_filters=32, kernel_size=5, pool_size=3, gru_units=32, dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.Conv1D(conv_filters, kernel_size=kernel_size, activation="relu"),
        layers.MaxPooling1D(pool_size=pool_size),
        layers.GRU(gru_units, dropout=dropout),
        layers.Dense(1)
    ])


def build_cnn_lstm(input_shape, conv_filters=32, kernel_size=5, pool_size=3, lstm_units=32, dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.Conv1D(conv_filters, kernel_size=kernel_size, activation="relu"),
        layers.MaxPooling1D(pool_size=pool_size),
        layers.LSTM(lstm_units, dropout=dropout),
        layers.Dense(1)
    ])


def build_bidirectional_lstm(input_shape, units=32, dropout=0.2):
    return keras.Sequential([
        layers.Input(shape=input_shape),
        layers.Bidirectional(layers.LSTM(units, dropout=dropout)),
        layers.Dense(1)
    ])

def build_model_from_config(config, input_shape):
    if config.model_type == "single_lstm":
        model = build_single_lstm(
            input_shape=input_shape,
            units=config.units1,
            dropout=config.dropout,
            recurrent_dropout=config.recurrent_dropout
        )
    elif config.model_type == "single_gru":
        model = build_single_gru(
            input_shape=input_shape,
            units=config.units1,
            dropout=config.dropout,
            recurrent_dropout=config.recurrent_dropout
        )
    elif config.model_type == "stacked_lstm":
        model = build_stacked_lstm(
            input_shape=input_shape,
            units1=config.units1,
            units2=config.units2,
            dropout=config.dropout,
            recurrent_dropout=config.recurrent_dropout
        )
    elif config.model_type == "stacked_gru":
        model = build_stacked_gru(
            input_shape=input_shape,
            units1=config.units1,
            units2=config.units2,
            dropout=config.dropout,
            recurrent_dropout=config.recurrent_dropout
        )
    elif config.model_type == "cnn_gru":
        model = build_cnn_gru(
            input_shape=input_shape,
            conv_filters=config.conv_filters,
            kernel_size=config.kernel_size,
            pool_size=config.pool_size,
            gru_units=config.units1,
            dropout=config.dropout
        )
    elif config.model_type == "cnn_lstm":
        model = build_cnn_lstm(
            input_shape=input_shape,
            conv_filters=config.conv_filters,
            kernel_size=config.kernel_size,
            pool_size=config.pool_size,
            lstm_units=config.units1,
            dropout=config.dropout
        )
    elif config.model_type == "bidirectional_lstm":
        model = build_bidirectional_lstm(
            input_shape=input_shape,
            units=config.units1,
            dropout=config.dropout
        )
    else:
        raise ValueError(f"Unknown model_type: {config.model_type}")

    model.compile(
        optimizer=keras.optimizers.Adam(learning_rate=LEARNING_RATE),
        loss="mse",
        metrics=["mae"]
    )
    return model

train_scaled, val_scaled, test_scaled, mean, std = standardize(train_df, val_df, test_df)

results = []
histories = {}
predictions = {}

for config in EXPERIMENTS:
    print("\n" + "=" * 80)
    print("Running:", config.name)
    print("=" * 80)

    train_ds = make_dataset(
        train_scaled, TARGET_COL,
        config.sequence_length, config.sampling_rate, config.delay_rows,
        BATCH_SIZE, True
    )
    val_ds = make_dataset(
        val_scaled, TARGET_COL,
        config.sequence_length, config.sampling_rate, config.delay_rows,
        BATCH_SIZE, False
    )
    test_ds = make_dataset(
        test_scaled, TARGET_COL,
        config.sequence_length, config.sampling_rate, config.delay_rows,
        BATCH_SIZE, False
    )

    for x_batch, _ in train_ds.take(1):
        input_shape = x_batch.shape[1:]

    baseline_val = baseline_mae_original(
        val_scaled, TARGET_COL,
        config.sequence_length, config.sampling_rate, config.delay_rows,
        mean, std
    )

    baseline_test = baseline_mae_original(
        test_scaled, TARGET_COL,
        config.sequence_length, config.sampling_rate, config.delay_rows,
        mean, std
    )

    model = build_model_from_config(config, input_shape)

    early_stop = keras.callbacks.EarlyStopping(
        monitor="val_mae",
        patience=PATIENCE,
        restore_best_weights=True
    )

    history = model.fit(
        train_ds,
        validation_data=val_ds,
        epochs=EPOCHS,
        callbacks=[early_stop],
        verbose=1
    )

    best_val_mae_scaled = float(np.min(history.history["val_mae"]))
    _, test_mae_scaled = model.evaluate(test_ds, verbose=0)

    val_mae_original, _, _ = evaluate_model_original_mae(
        model, val_ds, mean, std, TARGET_COL
    )
    test_mae_original, y_true_test, y_pred_test = evaluate_model_original_mae(
        model, test_ds, mean, std, TARGET_COL
    )

    results.append({
        "name": config.name,
        "model_type": config.model_type,
        "delay_rows": config.delay_rows,
        "units1": config.units1,
        "units2": config.units2,
        "recurrent_dropout": config.recurrent_dropout,
        "best_val_mae_scaled": best_val_mae_scaled,
        "val_mae_original": float(val_mae_original),
        "test_mae_scaled": float(test_mae_scaled),
        "test_mae_original": float(test_mae_original),
        "baseline_val_mae_original": float(baseline_val),
        "baseline_test_mae_original": float(baseline_test),
        "beats_baseline_val": bool(val_mae_original < baseline_val),
        "beats_baseline_test": bool(test_mae_original < baseline_test),
        "epochs_trained": len(history.history["loss"])
    })

    histories[config.name] = history
    predictions[config.name] = (y_true_test, y_pred_test)

from tensorflow import keras
from tensorflow.keras import layers

# Define parameters for the new CNN_GRU model's dataset
# Using parameters similar to the existing CNN_GRU experiment for consistency
SEQUENCE_LENGTH = 120
SAMPLING_RATE = 6
DELAY_ROWS = 144

# Create datasets for the new model
train_dataset = make_dataset(
    train_scaled, TARGET_COL,
    SEQUENCE_LENGTH, SAMPLING_RATE, DELAY_ROWS,
    BATCH_SIZE, True
)
val_dataset = make_dataset(
    val_scaled, TARGET_COL,
    SEQUENCE_LENGTH, SAMPLING_RATE, DELAY_ROWS,
    BATCH_SIZE, False
)
test_dataset = make_dataset(
    test_scaled, TARGET_COL,
    SEQUENCE_LENGTH, SAMPLING_RATE, DELAY_ROWS,
    BATCH_SIZE, False
)

# Determine input shape for the model
for x_batch, _ in train_dataset.take(1):
    input_shape = x_batch.shape[1:]

# Conv1D + GRU model
conv_gru_model = keras.Sequential([
    layers.Input(shape=input_shape), # Added Input layer with dynamic input_shape
    layers.Conv1D(32, 5, activation="relu"),
    layers.MaxPooling1D(3),
    layers.Conv1D(32, 5, activation="relu"),
    layers.GRU(32),
    layers.Dense(1)
])

conv_gru_model.compile(
    optimizer=keras.optimizers.Adam(),
    loss="mse",
    metrics=["mae"]
)

callbacks = [
    keras.callbacks.EarlyStopping(monitor="val_loss", patience=3)
]

history_conv_gru = conv_gru_model.fit(
    train_dataset,
    epochs=20,
    validation_data=val_dataset,
    callbacks=callbacks
)


val_mae_scaled = conv_gru_model.evaluate(val_dataset, verbose=0)[1]
test_mae_scaled = conv_gru_model.evaluate(test_dataset, verbose=0)[1]

# Evaluate MAE on original scale
val_mae_original, _, _ = evaluate_model_original_mae(
